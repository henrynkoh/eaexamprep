4. 최적화 전략

4.1 토큰 효율화 기법

from deepseek import Optimizer

def optimize_prompt(prompt):
    return Optimizer().compress(
        text=prompt,
        target_ratio=0.7,
        preserve_keywords=['분석', '요약']
    )

4.2 병렬 처리 예제

const { DeepSeek } = require('deepseek-api');
const parallelRequests = async (queries) => {
  return Promise.all(
    queries.map(query =>
      DeepSeek.chat.completions.create({
        model: "deepseek-r1",
        messages: [{role: "user", content: query}]
      })
    )
  );
};
